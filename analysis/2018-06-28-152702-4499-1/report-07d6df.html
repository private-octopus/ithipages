310static uint32_t xorshift_state = 0xdeadbeef;
311
312static uint32_t xorshift32()
313{
314 /* Algorithm "xor" from p. 4 of Marsaglia, "Xorshift RNGs" */
315 uint32_t x = xorshift_state;
316 x ^= x << 13;
317 x ^= x >> 17;
318 x ^= x << 5;
319 xorshift_state = x;
320 return x;
321}

323static uint32_t random_uniform(uint32_t range_max)
324{
325 uint32_t delta = 0xFFFFFFFF % range_max;
326 uint32_t rnd;
327
328 do {
329 rnd = xorshift32();
330 } while (rnd < delta);
331
332 return rnd % range_max;
333}

335#define RANDOM_TLD_LENGTH16 16

337static char * random_tld(char n[RANDOM_TLD_LENGTH16 + 1])
338{
339 for (int i = 0; i < RANDOM_TLD_LENGTH16; i++) {
340 n[i] = 'A' + random_uniform(26);
341 }
342 n[RANDOM_TLD_LENGTH16] = 0;
343
344 return n;

347bool TldCountTest::DoTest()
348{
349 LruHash<TldAsKey> tldStringUsage;
350 BinHash<DnsHashEntry> hashTable;
351 int count_per_string[nbTargetNames+1];
352 const int initial_count = 0x10000;
353 const int max_hash_size = 0x8000;
354 const int max_delta_count = 0x80;
355 const int max_tld_leakage_count = 0x80;
356 int current_count = initial_count;
357 int total_keys = 0;
358 int dropped_keys = 0;
359 int kept_keys = 0;
360 int expected_count = 0;
361 int hashed_count = 0;
362 int hashed_total = 0;
363 bool ret = true;
364 size_t * rand_table = NULL__null;
365
366 fill_initial_table(&hashTable);
367
368 for (size_t i = 1; i <= nbTargetNames; i++) {
369 total_keys += current_count;
370 if (i <= max_tld_leakage_count)
371 {
372 expected_count += current_count;
373 }
374 count_per_string[i] = current_count;
375 if (current_count >= 2) {
376 current_count >>= 1;
377 }
378 }
379
380 count_per_string[0] = 3*total_keys;
381 total_keys *= 4;
382
383 /* Simulate random arrivals */
384 rand_table = new size_t[total_keys];
385
386 if (rand_table == NULL__null) {
387 ret = false;
388 TEST_LOG("Tld count test, cannot allocate rand_table[%d]\n", total_keys);
389 } else {
390 int table_rank = 0;
391 /* Build a list of ID with desired size and repetition */
392 for (size_t i = 0; i <= nbTargetNames; i++) {
393 for (int c = 0; c < count_per_string[i] && table_rank < total_keys; c++) {
394 rand_table[table_rank++] = i;
395 }
396 }
397
398 /* Random shuffle of the list to simulate random arrivals */
399 for (int r = total_keys - 1; r >= 1; r--) {
400 int x = random_uniform((uint32_t)(r + 1));
401
402 if (x != r) {
403 size_t rtx = rand_table[x];
404 rand_table[x] = rand_table[r];
405 rand_table[r] = rtx;
406 }
407 }
408
409 /* Simulate insertion */
410 for (int i = 0; i < total_keys; i++)
411 {
412 bool stored = false;
413 char rand_name[RANDOM_TLD_LENGTH16 + 1];
414 char const * tld = (rand_table[i] == 0)? random_tld(rand_name): TargetNames[rand_table[i]-1];
415 TldAsKey key((uint8_t *)tld, strlen(tld));
416 TldAsKey * inserted = NULL__null;
417 DnsHashEntry counter_entry;
418
419 counter_entry.registry_id = REGISTRY_DNS_TLD_Usage_Count39;
420 counter_entry.key_type = 0;
421 counter_entry.key_length = sizeof(uint32_t);
422 counter_entry.key_number = 0;
423 counter_entry.count = 1;
424
425 (void)hashTable.InsertOrAdd(&counter_entry, true, &stored);
426
427 if (tldStringUsage.GetCount() >= max_hash_size)
428 {
429 TldAsKey * removed = tldStringUsage.RemoveLRU();
430 if (removed != NULL__null)
431 {
432 dropped_keys += removed->count;
433 delete removed;
434 }
435 }
436 inserted = tldStringUsage.InsertOrAdd(&key, true, &stored);
437 if (inserted != NULL__null && rand_table[i] > 0 && (int)inserted->count > count_per_string[rand_table[i]]) {
438 TEST_LOG("TLD %s, count %d instead of %d\n",
439 TargetNames[rand_table[i] - 1], inserted->count, count_per_string[rand_table[i]]);
440 }
441 }
442
443 /* Simulate extraction */
444 if (tldStringUsage.GetCount() > 0) {
445 TldAsKey *tld_entry;
446 std::vector<TldAsKey *> lines(tldStringUsage.GetCount());
447 int vector_index = 0;
448 uint32_t export_count = 0;
449
450
451 for (uint32_t i = 0; i < tldStringUsage.GetSize(); i++)
452 {
453 tld_entry = tldStringUsage.GetEntry(i);
454
455 while (tld_entry != NULL__null)
456 {
457 int tld_index = GetTldIndex(tld_entry->tld, tld_entry->tld_len);
458 if (tld_index >= 0) {
459 if (tld_index < max_tld_leakage_count &&
460 count_per_string[tld_index + 1] != (int)tld_entry->count) {
461 TEST_LOG("TLD %s, count %d instead of %d\n",
462 TargetNames[tld_index], tld_entry->count, count_per_string[tld_index + 1]);
463 if ((int)tld_entry->count > count_per_string[tld_index + 1]) {
464 ret = false;
465 }
466 }
467 }
468
469 lines[vector_index] = tld_entry;
470 vector_index++;
471 tld_entry = tld_entry->HashNext;
472 }
473 }
474
475 std::sort(lines.begin(), lines.end(), TldAsKey::CompareTldEntries);
476
477 /* Retain the N most interesting values */
478 for (size_t i = 0; i < lines.size(); i++)
479 {
480 if (export_count < max_tld_leakage_count)
481 {
482 kept_keys += lines[i]->count;
483 export_count++;
484 int tld_index = GetTldIndex(lines[i]->tld, lines[i]->tld_len);
485 DnsHashEntry key;
486 bool stored = false;
487
488 key.count = lines[i]->count;
489 key.registry_id = REGISTRY_DNS_Frequent_TLD_Usage38;
490 key.key_length = lines[i]->tld_len;
491 key.key_type = 1; /* string */
492 memcpy(key.key_value, lines[i]->tld, lines[i]->tld_len);
493 key.key_value[lines[i]->tld_len] = 0;
494
495 (void)hashTable.InsertOrAdd(&key, true, &stored);
496
497 if (tld_index < 0) {
498 if (lines[i]->count > 1) {
499 TEST_LOG("rank %d, random name, length %d, count %d\n",
500 i, lines[i]->tld_len, lines[i]->count);
501 }
502 } else {
503 if (count_per_string[tld_index + 1] != (int)lines[i]->count) {
504 TEST_LOG("rank %d, %s, count %d instead of %d\n",
505 i, TargetNames[tld_index], lines[i]->count, count_per_string[tld_index + 1]);
506 if ((int)lines[i]->count > count_per_string[tld_index + 1]) {
507 ret = false;
508 }
509 }
510 }
511 }
512 else
513 {
514 dropped_keys += lines[i]->count;
515 }
516 }
517 }
518
519 /* Verify count */
520 ret &= ((kept_keys + dropped_keys) == total_keys);
521
522 if (!ret) {
523 TEST_LOG("Tld count test, kept_keys + dropped_keys: %d + %d = %d != total: %d\n",
524 kept_keys, dropped_keys, kept_keys + dropped_keys, total_keys);
525 } else {
526 int delta_count = expected_count - kept_keys;
527
528 ret = (delta_count <= max_delta_count) && (-delta_count <= max_delta_count);
529 if (!ret) {
530 TEST_LOG("Tld count test, expected_count: %d >> kept_keys: %d\n",
531 expected_count, kept_keys);
532 }
533 }
534
535 if (ret){
536 /* Verify the values from the hash table */
537 DnsHashEntry *entry;
538
539 for (uint32_t i = 0; i < hashTable.GetSize(); i++)
540 {
541 entry = hashTable.GetEntry(i);
542
543 while (entry != NULL__null)
544 {
545 if (entry->registry_id == REGISTRY_DNS_Frequent_TLD_Usage38) {
546 hashed_count += entry->count;
547 }
548 else if (entry->registry_id == REGISTRY_DNS_TLD_Usage_Count39) {
549 hashed_total = entry->count;
550 }
551
552 entry = entry->HashNext;
553 }
554 }
555
556 if (hashed_count != kept_keys) {
557 TEST_LOG("Tld sum count after hash: %d != kept_keys: %d\n",
558 hashed_count, kept_keys);
559 ret = false;
560 } else if (hashed_total != total_keys) {
561 TEST_LOG("Tld total from hash: %d != total_keys: %d\n",
562 hashed_total, total_keys);
563 ret = false;
564 }
565 }
566 }
567 return ret;
568}

570int TldCountTest::GetTldIndex(uint8_t * tld, size_t tld_len)
571{
572 int low = -1;
573 int high = nbTargetNames;
574 int middle;
575 int found = -1;
576
577 while (found == -1 && low < high) {
578 uint8_t * r;
579 size_t r_len;
580 int cmp = 0;
581
582 middle = low + (high - low) / 2;
583
584 if (middle == low || middle == high) {
585 break;
586 }
587
588 r = (uint8_t *) TargetNames[middle];
589 r_len = strlen(TargetNames[middle]);
590
591
592 for (size_t i = 0; cmp == 0 && i < tld_len; i++) {
593 if (i > r_len || tld[i] > r[i]) {
594 cmp = 1; /* Tld is below middle */
595 }
596 else if (tld[i] < r[i]) {
597 cmp = -1; /* Tld is above middle */
598 }
599 }
600 if (cmp == 0 && r_len > tld_len) {
601 cmp = -1; /* name is longer, tld is below it */
602 }
603
604 switch (cmp) {
605 case -1:
606 high = middle;
607 break;
608 case 0:
609 found = middle;
610 break;
611 case 1:
612 low = middle;
613 break;
614 }
615 }
616
617 return found;
