314int DnsStats::SubmitQuery(uint8_t * packet, uint32_t length, uint32_t start, bool is_response)
315{
316 int rrclass = 0;
317 int rrtype = 0;
318 uint32_t name_start = start;
319
320 /* TODO: if we decide to tabulate parameters in queries, will need
321 * to check "is_response" value and controlling flag */
322 UNREFERENCED_PARAMETER(is_response)(void)(is_response);
323
324 start = SubmitName(packet, length, start, false);
325
326 if (start + 4 <= length)
327 {
328 rrtype = (packet[start] << 8) | packet[start + 1];
329 rrclass = (packet[start + 2] << 8) | packet[start + 3];
330 start += 4;
331
332 if (dnsstat_flags&dnsStateFlagCountQueryParms)
333 {
334 SubmitRegistryNumber(REGISTRY_DNS_Q_CLASSES15, rrclass);
335 SubmitRegistryNumber(REGISTRY_DNS_Q_RRType16, rrtype);
336 }
337
338 if (dnsstat_flags&dnsStateFlagCountUnderlinedNames)
339 {
340 if (rrtype == DnsRtype_TXT)
341 {
342 SubmitRegistryString(REGISTRY_DNS_txt_underline26, 3, (uint8_t *) "TXT");
343 CheckForUnderline(packet, length, name_start);
344 }
345 }
346 }
347 else
348 {
349 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
350 start = length;
351 }
352
353 return start;
354}

356int DnsStats::SubmitRecord(uint8_t * packet, uint32_t length, uint32_t start,
357 uint32_t * e_rcode, uint32_t * e_length, bool is_response)
358{
359 int rrtype = 0;
360 int rrclass = 0;
361 unsigned int ttl = 0;
362 int ldata = 0;
363 int name_start = start;
364
365 record_count++;
366
367 /* Labels are only tabulated in responses, to avoid polluting data with erroneous packets */
368 start = SubmitName(packet, length, start, is_response);
369
370 if ((start + 10) > length)
371 {
372 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
373 start = length;
374 }
375 else
376 {
377 rrtype = (packet[start] << 8) | packet[start + 1];
378 rrclass = (packet[start + 2] << 8) | packet[start + 3];
379 ttl = (packet[start + 4] << 24) | (packet[start + 5] << 16)
380 | (packet[start + 6] << 8) | packet[start + 7];
381 ldata = (packet[start + 8] << 8) | packet[start + 9];
382
383 if (start + ldata + 10 > length)
384 {
385 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
386 start = length;
387 }
388 else
389 {
390 if (ldata > 0 || rrtype == DnsRtype_OPT)
391 {
392 /* only record rrtypes and rrclass if valid response */
393 if (rrtype != DnsRtype_OPT)
394 {
395 if (is_response)
396 {
397 SubmitRegistryNumber(REGISTRY_DNS_CLASSES1, rrclass);
398 }
399 else if (dnsstat_flags&dnsStateFlagCountQueryParms)
400 {
401 SubmitRegistryNumber(REGISTRY_DNS_Q_CLASSES15, rrclass);
402 }
403 }
404 else
405 {
406 /* document the extended length */
407 if (e_length != NULL__null)
408 {
409 *e_length = rrclass;
410 }
411 }
412
413 if (is_response)
414 {
415 SubmitRegistryNumber(REGISTRY_DNS_RRType2, rrtype);
416 }
417 else if (dnsstat_flags&dnsStateFlagCountQueryParms)
418 {
419 SubmitRegistryNumber(REGISTRY_DNS_Q_RRType16, rrtype);
420 }
421
422 /* For records of type RRSIG, NSEC, NSEC3, DNSKEY, DS,
423 * mark the domain as supporting DNSSEC */
424 if (dnssec_name_index == 0 && (
425 rrtype == DnsRtype_DNSKEY ||
426 rrtype == DnsRtype_RRSIG ||
427 rrtype == DnsRtype_NSEC ||
428 rrtype == DnsRtype_NSEC3 ||
429 rrtype == DnsRtype_DS)) {
430 dnssec_name_index = name_start;
431 }
432
433 /* Further parsing for OPT, DNSKEY, RRSIG, DS,
434 * and maybe also AFSDB, NSEC3, DHCID, RSYNC types */
435 switch (rrtype)
436 {
437 case (int)DnsRtype_OPT:
438 SubmitOPTRecord(ttl, &packet[start + 10], ldata, e_rcode);
439 break;
440 case (int)DnsRtype_DNSKEY:
441 SubmitKeyRecord(&packet[start + 10], ldata);
442 break;
443 case (int)DnsRtype_RRSIG:
444 SubmitRRSIGRecord(&packet[start + 10], ldata);
445 break;
446 case (int)DnsRtype_DS:
447 SubmitDSRecord(&packet[start + 10], ldata);
448 break;
449 case (int)DnsRtype_TLSA:
450 SubmitTLSARecord(&packet[start + 10], ldata);
451 break;
452 default:
453 break;
454 }
455 }
456
457 start += ldata + 10;
458 }
459 }
460
461 return start;
462}

464int DnsStats::SubmitName(uint8_t * packet, uint32_t length, uint32_t start, bool should_tabulate)
465{
466 uint32_t l = 0;
467
468 while (start < length)
469 {
470 l = packet[start];
471
472 if (l == 0)
473 {
474 /* end of parsing*/
475 if (should_tabulate)
476 {
477 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, 0);
478 }
479 start++;
480 break;
481 }
482 else if ((l & 0xC0) == 0xC0)
483 {
484 /* Name compression */
485 if (should_tabulate)
486 {
487 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, 0xC0);
488 }
489
490 if ((start + 2) > length)
491 {
492 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
493 start = length;
494 break;
495 }
496 else
497 {
498 start += 2;
499 break;
500 }
501 }
502 else if (l > 0x3F)
503 {
504 /* found an extension. Don't know how to parse it! */
505 error_flags |= DNS_REGISTRY_ERROR_LABEL(1<<7);
506 if (should_tabulate)
507 {
508 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, l);
509 }
510 start = length;
511 break;
512 }
513 else
514 {
515 /* regular name part. To do: tracking of underscore labels. */
516 if (should_tabulate)
517 {
518 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, 0);
519 }
520 if (start + l + 1 > length)
521 {
522 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
523 start = length;
524 break;
525 }
526 else
527 {
528 start += l + 1;
529 }
530 }
531 }
532
533 return start;

536void DnsStats::SubmitOPTRecord(uint32_t flags, uint8_t * content, uint32_t length, uint32_t * e_rcode)
537{
538 uint32_t current_index = 0;
539
540 /* Process the flags and rcodes */
541 if (e_rcode != NULL__null)
542 {
543 *e_rcode = (flags >> 24) & 0xFF;
544 }
545
546 /* Register whether the DO bit was set */
547 is_do_flag_set = (flags & (1<<15)) != 0;
548
549 /* Add flags to registration */
550 for (int i = 0; i < 16; i++)
551 {
552 if ((flags & (1 << i)) != 0)
553 {
554 SubmitRegistryNumber(REGISTRY_EDNS_Header_Flags10, 15 - i);
555 }
556 }
557
558 SubmitRegistryNumber(REGISTRY_EDNS_Version_number11, (flags >> 16) & 0xFF);
559
560 /* Find the options in the payload */
561 while (current_index + 4 <= length)
562 {
563 uint32_t o_code = (content[current_index] << 8) | content[current_index + 1];
564 uint32_t o_length = (content[current_index+2] << 8) | content[current_index + 3];
565 current_index += 4 + o_length;
566 SubmitRegistryNumber(REGISTRY_EDNS_OPT_CODE8, o_code);
567 }
568}

664int DnsStats::CheckForUnderline(uint8_t * packet, uint32_t length, uint32_t start)
665{
666 uint32_t l = 0;
667 uint32_t name_start = start;
668
669 while (start < length)
670 {
671 l = packet[start];
672
673 if (l == 0)
674 {
675 /* end of parsing*/
676 start++;
677 break;
678 }
679 else if ((l & 0xC0) == 0xC0)
680 {
681 if ((start + 2) > length)
682 {
683 /* error */
684 start = length;
685 break;
686 }
687 else
688 {
689 uint32_t new_start = ((l & 63) << 8) + packet[start + 1];
690
691 if (new_start < name_start)
692 {
693 (void)CheckForUnderline(packet, length, new_start);
694 }
695
696 start += 2;
697 break;
698 }
699 }
700 else if (l > 0x3F)
701 {
702 /* found an extension. Don't know how to parse it! */
703 start = length;
704 break;
705 }
706 else
707 {
708 /* Tracking of underscore labels. */
709 if (start + l + 1 > length)
710 {
711 /* format error */
712 start = length;
713 break;
714 }
715 else
716 {
717 if (l > 0 && packet[start + 1] == '_')
718 {
719 uint8_t underlined_name[64];
720 uint32_t flags;
721
722 NormalizeNamePart(l, &packet[start + 1], underlined_name, &flags);
723
724 if ((flags & 3) == 0)
725 {
726 SubmitRegistryString(REGISTRY_DNS_txt_underline26, l, underlined_name);
727 }
728 }
729 start += l + 1;
730 }
731 }
732 }
733
734 return start;
735}

797 break;
798 }
799 else
800 {
801 previous = start;

803 }
804 }

810int DnsStats::GetDnsName(uint8_t * packet, uint32_t length, uint32_t start,
811 uint8_t * name, size_t name_max, size_t * name_length)
812{
813 uint32_t l = 0;
814 uint32_t name_start = start;
815 uint32_t start_next = 0;
816 size_t name_index = 0;
817
818 while (start <length && name_index < name_max) {
819 l = packet[start];
820
821 if (l == 0)
822 {
823 /* end of parsing*/
824 start++;
825
826 if (start_next == 0) {
827 start_next = start;
828 }
829 break;
830 }
831 else if ((l & 0xC0) == 0xC0)
832 {
833 if ((start + 2) > length)
834 {
835 /* error */
836 start_next = length;
837 break;
838 }
839 else
840 {
841 uint32_t new_start = ((l & 63) << 8) + packet[start + 1];
842
843 if (new_start < name_start)
844 {
845 if (start_next == 0) {
846 start_next = start + 2;
847 }
848 start = new_start;
849 } else {
850 /* Basic restriction to avoid name decoding loops */
851 name_index = 0;
852 start_next = length;
853 break;
854 }
855 }
856 }
857 else if (l > 0x3F)
858 {
859 /* found an extension. Don't know how to parse it! */
860 name_index = 0;
861 start_next = length;
862 break;
863 }
864 else
865 {
866 /* add a label to the name. */
867 if (start + l + 1 > length ||
868 name_index + l + 2 > name_max)
869 {
870 /* format error */
871 name_index = 0;
872 start_next = length;
873 break;
874 }
875 else
876 {
877 uint32_t flags;
878 if (name_index > 0) {
879 name[name_index++] = '.';
880 }
881
882 NormalizeNamePart(l, &packet[start + 1], name + name_index, &flags);
883
884 if ((flags & 3) == 0) {
885 start += l + 1;
886 name_index += l;
887 } else {
888 /* format error */
889 name_index = 0;
890 start_next = length;
891 break;
892 }
893 }
894 }
895 }
896
897 name[name_index] = 0;
898
899 *name_length = name_index;
900
901 return start_next;
902
903}

905void DnsStats::NormalizeNamePart(uint32_t length, uint8_t * value,
906 uint8_t * normalized, uint32_t * flags)
907{
908 bool has_letter = false;
909 bool has_number = false;
910 bool has_special = false;
911 bool has_dash = false;
912 bool has_non_ascii = false;
913
914 for (uint32_t i = 0; i < length; i++)
915 {
916 uint8_t c = value[i];
917 if (c >= 'A' && c <= 'Z')
918 {
919 has_letter = true;
920 }
921 else if (c >= 'a' && c <= 'z')
922 {
923 has_letter = true;
924 }
925 else if (c >= '0' && c <= '9')
926 {
927 has_number = true;
928 }
929 else if (c == '-' || c == '_')
930 {
931 has_dash = true;
932 }
933 else if (c > 127)
934 {
935 has_non_ascii = true;
936 }
937 else if (c <= ' ' || c == 127 || c == '"' || c == ',')
938 {
939 c = '?';
940 has_special = true;
941 }
942 else
943 {
944 has_special = true;
945 }
946 normalized[i] = c;
947 }
948 normalized[length] = 0;
949
950 if (flags != NULL__null)
951 {
952 *flags = 0;
953
954 if (has_non_ascii)
955 {
956 *flags += 1;
957 }
958 else if (has_special)
959 {
960 *flags += 2;
961 }
962
963 if (has_letter)
964 {
965 *flags += 64;
966 }
967 if (has_number)
968 {
969 *flags += 128;
970 }
971 if (has_dash)
972 {
973 *flags += 256;
974 }
975 }
976}

1306 uint8_t * source_addr, size_t source_addr_length,
1307 uint8_t * dest_addr, size_t dest_addr_length)
1308{
1309 bool is_response;
1310
1311 bool has_header = true;
1312 uint32_t flags = 0;
1313 uint32_t opcode = 0;
1314 uint32_t rcode = 0;
1315 uint32_t e_rcode = 0;
1316 uint32_t qdcount = 0;
1317 uint32_t ancount = 0;
1318 uint32_t nscount = 0;
1319 uint32_t arcount = 0;
1320 uint32_t parse_index = 0;
1321 uint32_t e_length = 512;
1322 bool unfiltered = false;
1323
1324 error_flags = 0;
1325 is_do_flag_set = false;
1326 dnssec_name_index = 0;
1327
1328 if (rootAddresses.GetCount() == 0)
1329 {
1330 rootAddresses.SetList(DefaultRootAddresses, sizeof(DefaultRootAddresses) / sizeof(char const *));
1331 }
1332
1333 if (length < 12)
1334 {
1335 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
1336 parse_index = length;
1337 has_header = false;
1338 }
1339 else
1340 {
1341 is_response = ((packet[2] & 128) != 0);
1342
1343 if (is_response)
1344 {
1345 unfiltered = CheckAddress(dest_addr, dest_addr_length);
1346 }
1347 else
1348 {
1349 unfiltered = CheckAddress(source_addr, source_addr_length);
1350 }
1351 }
1352
1353 if (unfiltered)
1354 {
1355 if (is_response)
1356 {
1357 response_count++;
1358 }
1359 else
1360 {
1361 query_count++;
1362 }
1363
1364 flags = ((packet[2] & 7) << 4) | ((packet[3] & 15) >> 4);
1365 opcode = (packet[2] >> 3) & 15;
1366 rcode = (packet[3] & 15);
1367 qdcount = (packet[4] << 8) | packet[5];
1368 ancount = (packet[6] << 8) | packet[7];
1369 nscount = (packet[8] << 8) | packet[9];
1370 arcount = (packet[10] << 8) | packet[11];
1371
1372 SubmitRegistryNumber(REGISTRY_DNS_OpCodes3, opcode);
1373
1374 if (is_response && opcode == DNS_OPCODE_QUERY0)
1375 {
1376
1377 uint32_t tld_offset = 0;
1378 bool gotTld = GetTLD(packet, length, 12, &tld_offset);
1379
1380 if (gotTld)
1381 {
1382 SetToUpperCase(packet + tld_offset + 1, packet[tld_offset]);
1383 }
1384
1385 if (rootAddresses.IsInList(source_addr, source_addr_length))
1386 {
1387 /* Perform statistics on root traffic */
1388
1389 SubmitRegistryNumber(REGISTRY_DNS_root_QR27, rcode);
1390
1391 if (gotTld)
1392 {
1393 if (rcode == DNS_RCODE_NXDOMAIN3 && packet[tld_offset] != 0)
1394 {
1395 /* Analysis of domain leakage */
1396 if (IsRfc6761Tld(packet + tld_offset + 1, packet[tld_offset]))
1397 {
1398 SubmitRegistryString(REGISTRY_DNS_RFC6761TLD30, packet[tld_offset], packet + tld_offset + 1);
1399 }
1400 else
1401 {
1402 /* Insert in leakage table */
1403 TldAsKey key(packet + tld_offset + 1, packet[tld_offset]);
1404 bool stored = false;
1405 (void)tldLeakage.InsertOrAdd(&key, true, &stored);
1406
1407 /* TODO: If full enough, remove the LRU, and account for it in the patterns catalog */
1408 if (tldLeakage.GetCount() > max_tld_leakage_table_count)
1409 {
1410 TldAsKey * removed = tldLeakage.RemoveLRU();
1411 if (removed != NULL__null)
1412 {
1413 /* Add count of leaks by length -- should replace by pattern match later */
1414 SubmitRegistryNumber(REGISTRY_DNS_LeakByLength28, (uint32_t) removed->tld_len);
1415
1416 delete removed;
1417 }
1418 }
1419 }
1420 }
1421 else if (rcode == DNS_RCODE_NOERROR0)
1422 {
1423 /* Analysis of useless traffic to the root */
1424 TldAddressAsKey key(dest_addr, dest_addr_length, packet + tld_offset + 1, packet[tld_offset]);
1425
1426 if (queryUsage.GetCount() >= max_query_usage_count)
1427 {
1428 /* Table is full. Just keep counting the transactions that are present */
1429 TldAddressAsKey * present = queryUsage.Retrieve(&key);
1430 if (present != NULL__null)
1431 {
1432 present->count++;
1433 SubmitRegistryNumber(REGISTRY_DNS_UsefulQueries31, 0);
1434 }
1435 }
1436 else
1437 {
1438 bool stored = false;
1439 (void)queryUsage.InsertOrAdd(&key, true, &stored);
1440
1441 SubmitRegistryNumber(REGISTRY_DNS_UsefulQueries31, (stored) ? 1 : 0);
1442 }
1443
1444 if (dnsstat_flags&dnsStateFlagCountTld)
1445 {
1446 SubmitRegistryString(REGISTRY_TLD_response23, packet[tld_offset], packet + tld_offset + 1);
1447 }
1448 }
1449 }
1450 }
1451 else if (gotTld)
1452 {
1453 /* Perform statistics on user traffic */
1454 TldAsKey key(packet + tld_offset + 1, packet[tld_offset]);
1455
1456 /* Check whether this TLD is in the registered list */
1457 if (registeredTld.GetCount() == 0)
1458 {
1459 LoadRegisteredTLD_from_memory();
1460 }
1461
1462 if (registeredTld.Retrieve(&key) != NULL__null)
1463 {
1464 /* This is a registered TLD */
1465 SubmitRegistryNumber(REGISTRY_DNS_TLD_Usage_Count39, 1);
1466 if ((dnsstat_flags&dnsStateFlagListTldUsed) != 0)
1467 {
1468 SubmitRegistryString(REGISTRY_DNS_Tld_Usage36, packet[tld_offset], packet + tld_offset + 1);
1469 }
1470 }
1471 else
1472 {
1473 /* Keep a count */
1474 SubmitRegistryNumber(REGISTRY_DNS_TLD_Usage_Count39, 0);
1475
1476 if (IsRfc6761Tld(packet + tld_offset + 1, packet[tld_offset]))
1477 {
1478 SubmitRegistryString(REGISTRY_DNS_RFC6761_Usage37, packet[tld_offset], packet + tld_offset + 1);
1479 }
1480 else
1481 {
1482 bool stored = false;
1483
1484 if (tldStringUsage.GetCount() >= max_tld_string_usage_count)
1485 {
1486 TldAsKey * removed = tldStringUsage.RemoveLRU();
1487 if (removed != NULL__null)
1488 {
1489 delete removed;
1490 }
1491 }
1492
1493 tldStringUsage.InsertOrAdd(&key, true, &stored);
1494 }
1495 }
1496 }
1497 }
1498
1499 for (uint32_t i = 0; i < 7; i++)
1500 {
1501 if ((flags & (1 << i)) != 0)
1502 {
1503 SubmitRegistryNumber(REGISTRY_DNS_Header_Flags9, i);
1504 }
1505 }
1506
1507 parse_index = 12;
1508 }
1509
1510 for (uint32_t i = 0; i < qdcount; i++)
1511 {
1512 if (parse_index >= length)
1513 {
1514 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
1515 }
1516 else
1517 {
1518 parse_index = SubmitQuery(packet, length, parse_index, is_response);
1519 }
1520 }
1521
1522 for (uint32_t i = 0; i < ancount; i++)
1523 {
1524 if (parse_index >= length)
1525 {
1526 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
1527 }
1528 else
1529 {
1530 parse_index = SubmitRecord(packet, length, parse_index, NULL__null, NULL__null, is_response);
1531 }
1532 }
1533
1534 for (uint32_t i = 0; i < nscount; i++)
1535 {
1536 if (parse_index >= length)
1537 {
1538 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
1539 }
1540 else
1541 {
1542 parse_index = SubmitRecord(packet, length, parse_index, NULL__null, NULL__null, is_response);
1543 }
1544 }
1545
1546 for (uint32_t i = 0; i < arcount; i++)
1547 {
1548 if (parse_index >= length)
1549 {
1550 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
1551 }
1552 else
1553 {
1554 parse_index = SubmitRecord(packet, length, parse_index, &e_rcode, &e_length, is_response);
1555 }
1556 }
1557
1558 if (has_header)
1559 {
1560 rcode |= (e_rcode << 4);
1561 SubmitRegistryNumber(REGISTRY_DNS_RCODES4, rcode);
1562 }
1563

1565 {

1567 {
1568 SubmitRegistryNumber(REGISTRY_DNS_Response_Size20, length);

1570 {
1571 SubmitRegistryNumber(REGISTRY_DNS_TC_length21, e_length);
1572 }
1573 }
1574 else
1575 {
1576 SubmitRegistryNumber(REGISTRY_DNS_Query_Size19, length);
1577 SubmitRegistryNumber(REGISTRY_EDNS_Packet_Size18, e_length);
1578 }
1579 }
1580
1581 SubmitRegistryNumber(REGISTRY_DNS_error_flag24, error_flags);
1582
1583 if (has_header && is_response && opcode == DNS_OPCODE_QUERY0 &&
1584 rcode == DNS_RCODE_NOERROR0 && error_flags == 0 &&
1585 !rootAddresses.IsInList(source_addr, source_addr_length)) {
1586 /* Do not perform client statistic on root traffic, but do it
1587 * for all other sources of traffic */
1588 RegisterDnssecUsageByAddress(dest_addr, dest_addr_length);
1589
1590 if (is_do_flag_set) {
1591 if (dnssec_name_index == 0) {
1592 RegisterDnssecUsageByName(packet, length, 12, false);
1593 } else {
1594 RegisterDnssecUsageByName(packet, length, dnssec_name_index, true);
