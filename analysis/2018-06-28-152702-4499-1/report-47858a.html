70static double Tld_Dga_Eval_P(size_t len, uint8_t * val)
71{
72 size_t count[5];
73 double p = 1.0;
74
75 memset(count, 0, sizeof(count));
76
77 for (size_t i=0; i<len; i++)
78 {
79 uint8_t char_index = val[i] - 10;
80 p *= 26.0*char_proba[char_index];
81 }
82
83 return p;
84}

86static double Tld_Dga_Eval_C_U(size_t len, uint8_t * val)
87{
88 size_t count[5];
89 double klt = 0;
90
91 memset(count, 0, sizeof(count));
92
93 for (size_t i = 0; i < len; i++)
94 {
95 uint8_t char_group_index = char_group[val[i] - 10];
96 count[char_group_index] ++;
97 }
98
99 for (int j = 0; j < 5; j++)
100 {
101 double p = ((double)count[j] + (double)char_group_weight[j] / 26.0) / ((double)(len + 1));
102 double kl = p * log(26.0*p / double(char_group_weight[j]));
103 klt += kl;
104 }
105
106 return klt;

109static double Tld_Dga_Eval_T_U(size_t len, uint8_t * val)
110{
111 size_t count[4];
112 double klt = 0;
113 uint8_t previous_group = 2;
114
115 memset(count, 0, sizeof(count));
116
117 for (size_t i = 0; i<len; i++)
118 {
119 uint8_t t_group_index = transition_group[val[i] - 10];
120
121 if (previous_group < 2)
122 {
123 count[2 * previous_group + t_group_index] ++;

132 klt += kl;

226bool NamePattern::Preprocess(uint8_t * str, size_t len, uint8_t * val, uint32_t * char_pattern)
227{
228 bool ret = true;
229 uint32_t flags = name_pattern_alphanum |
230 name_pattern_alpha | name_pattern_hexa | name_pattern_num;
231
232 for (size_t i = 0; i < len; i++)
233 {
234 int x = str[i];
235 val[i] = 0;
236
237 if (x >= '0' && x <= '9')
238 {
239 val[i] = x - '0';
240 flags &= (name_pattern_alphanum | name_pattern_hexa | name_pattern_num);
241 }
242 else if (x >= 'a' && x <= 'f')
243 {
244 val[i] = 10 + x - 'a';
245 flags &= (name_pattern_alphanum | name_pattern_alpha | name_pattern_hexa);
246 }
247 else if (x >= 'g' && x <= 'z')
248 {
249 val[i] = 10 + x - 'a';
250 flags &= (name_pattern_alphanum | name_pattern_alpha);
251 }
252 else if (x >= 'A' && x <= 'F')
253 {
254 val[i] = 10 + x - 'A';
255 flags &= (name_pattern_alphanum | name_pattern_alpha | name_pattern_hexa);
256 }
257 else if (x >= 'G' && x <= 'Z')
258 {
259 val[i] = 10 + x - 'A';
260 flags &= (name_pattern_alphanum | name_pattern_alpha);
261 }
262 else if (x == '-')
263 {
264 val[i] = 36;
265 if (i == 0)
266 {
267 flags &= (name_pattern_num);
268 }
269 else
270 {
271 flags = 0;
272 }
273 }
274 else
275 {
276 flags = 0;
277 ret = false;
278 break;
279 }
280 }
281
282 if (ret && flags == 0 && len > 4)
283 {
284 if (val[0] == 10 + 'x' - 'a' &&
285 val[1] == 10 + 'n' - 'a' &&
286 val[2] == 36 && val[3] == 36)
287 {
288 flags |= name_pattern_idn;
289 }
290 }
291
292 *char_pattern = flags;
293
294 return ret;
295}
