215bool ithipublisher::ParseFileName(MetricFileHolder * file, const char * name, int metric_id)
216{
217 size_t ch_index = 0;
218 size_t char_after_sep_index = 0;
219 size_t name_len = strlen(name);
220 bool ret = name_len < sizeof(file->file_name);
221
222 /* Find the last separator in the file name */
223 if (ret)
224 {
225 while (name[ch_index] != 0)
226 {
227 if (name[ch_index] == ITHI_FILE_PATH_SEP"/"[0])
228 {
229 char_after_sep_index = ch_index + 1;
230 }
231 ch_index++;
232 }
233
234 /* Check that the name is well formed */
235 ret &= (char_after_sep_index + 17u) <= name_len;
236 }
237
238 if (ret)
239 {
240 ret = name[char_after_sep_index] == 'M' &&
241 name[char_after_sep_index + 1] == '0' + metric_id &&
242 name[char_after_sep_index + 2] == '-' &&
243 name[char_after_sep_index + 7] == '-' &&
244 name[char_after_sep_index + 10] == '-' &&
245 name[char_after_sep_index + 13] == '.' &&
246 name[char_after_sep_index + 14] == 'c' &&
247 name[char_after_sep_index + 15] == 's' &&
248 name[char_after_sep_index + 16] == 'v' &&
249 name[char_after_sep_index + 17] == 0;
250 }
251
252 if (ret)
253 {
254 char digits[5];
255 int val[3] = { 0, 0, 0 };
256 const int delta[3] = { 3, 8, 11 };
257 const int len[3] = { 4, 2, 2 };
258
259 for (int i = 0; i < 3; i++)
260 {
261 for (int j = 0; j < len[i]; j++)
262 {
263 digits[j] = name[char_after_sep_index + delta[i] + j];
264 ret &= (isdigit(digits[j]) != 0);
265 }
266 digits[len[i]] = 0;
267 val[i] = atoi(digits);
268 }
269
270 file->year = val[0];
271 file->month = val[1];
272 file->day = val[2];
273 }
274
275 if (ret)
276 {
277 memcpy(file->file_name, name, name_len + 1);
278 }
279
280 return ret;
