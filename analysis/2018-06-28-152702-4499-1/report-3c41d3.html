132static int check_execution_mode(ithi_tool_mode proposed, ithi_tool_mode * current)
133{
134 int exit_code = 0;
135 if (*current != ithi_mode_unspecified && proposed != 0)
136 {
137 fprintf(stderrstderr, "Can only specify one execution mode!\n");
138 exit_code = usage();
139 }
140 else
141 {
142 *current = proposed;
143 }
144
145 return exit_code;
146}
