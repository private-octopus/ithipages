472int DnsStats::SubmitQuery(uint8_t * packet, uint32_t length, uint32_t start, bool is_response, int * qclass, int * qtype)
473{
474 int rrclass = 0;
475 int rrtype = 0;
476 uint32_t name_start = start;
477
478 /* TODO: if we decide to tabulate parameters in queries, will need
479 * to check "is_response" value and controlling flag */
480 UNREFERENCED_PARAMETER(is_response)(void)(is_response);
481
482 start = SubmitName(packet, length, start, false);
483
484 if (start + 4 <= length)
485 {
486 rrtype = (packet[start] << 8) | packet[start + 1];
487 rrclass = (packet[start + 2] << 8) | packet[start + 3];
488 start += 4;
489 *qclass = rrclass;
490 *qtype = rrtype;
491
492 if (dnsstat_flags&dnsStateFlagCountQueryParms)
493 {
494 SubmitRegistryNumber(REGISTRY_DNS_Q_CLASSES15, rrclass);
495 SubmitRegistryNumber(REGISTRY_DNS_Q_RRType16, rrtype);
496 }
497
498 if (dnsstat_flags&dnsStateFlagCountUnderlinedNames)
499 {
500 if (rrtype == DnsRtype_TXT)
501 {
502 SubmitRegistryString(REGISTRY_DNS_txt_underline26, 3, (uint8_t *) "TXT");
503 CheckForUnderline(packet, length, name_start);
504 }
505 }
506 }
507 else
508 {
509 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
510 start = length;
511 }
512
513 return start;
514}

516int DnsStats::SubmitRecord(uint8_t * packet, uint32_t length, uint32_t start,
517 uint32_t * e_rcode, uint32_t * e_length, bool is_response)
518{
519 int rrtype = 0;
520 int rrclass = 0;
521 unsigned int ttl = 0;
522 int ldata = 0;
523 int name_start = start;
524
525 record_count++;
526
527 /* Labels are only tabulated in responses, to avoid polluting data with erroneous packets */
528 start = SubmitName(packet, length, start, is_response);
529
530 if ((start + 10) > length)
531 {
532 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
533 start = length;
534 }
535 else
536 {
537 rrtype = (packet[start] << 8) | packet[start + 1];
538 rrclass = (packet[start + 2] << 8) | packet[start + 3];
539 ttl = (packet[start + 4] << 24) | (packet[start + 5] << 16)
540 | (packet[start + 6] << 8) | packet[start + 7];
541 ldata = (packet[start + 8] << 8) | packet[start + 9];
542
543 if (start + ldata + 10 > length)
544 {
545 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
546 start = length;
547 }
548 else
549 {
550 if (ldata > 0 || rrtype == DnsRtype_OPT)
551 {
552 /* only record rrtypes and rrclass if valid response */
553 if (rrtype != DnsRtype_OPT)
554 {
555 if (is_response)
556 {
557 SubmitRegistryNumber(REGISTRY_DNS_CLASSES1, rrclass);
558 }
559 else if (dnsstat_flags&dnsStateFlagCountQueryParms)
560 {
561 SubmitRegistryNumber(REGISTRY_DNS_Q_CLASSES15, rrclass);
562 }
563 }
564 else
565 {
566 /* document the extended length */
567 if (e_length != NULL__null)
568 {
569 *e_length = rrclass;
570 }
571 }
572
573 if (is_response)
574 {
575 SubmitRegistryNumber(REGISTRY_DNS_RRType2, rrtype);
576 }
577 else if (dnsstat_flags&dnsStateFlagCountQueryParms)
578 {
579 SubmitRegistryNumber(REGISTRY_DNS_Q_RRType16, rrtype);
580 }
581
582 /* For records of type RRSIG, NSEC, NSEC3, DNSKEY, DS,
583 * mark the domain as supporting DNSSEC */
584 if (dnssec_name_index == 0 && (
585 rrtype == DnsRtype_DNSKEY ||
586 rrtype == DnsRtype_RRSIG ||
587 rrtype == DnsRtype_NSEC ||
588 rrtype == DnsRtype_NSEC3 ||
589 rrtype == DnsRtype_DS)) {
590 dnssec_name_index = name_start;
591 }
592
593 /* Further parsing for OPT, DNSKEY, RRSIG, DS,
594 * and maybe also AFSDB, NSEC3, DHCID, RSYNC types */
595 switch (rrtype)
596 {
597 case (int)DnsRtype_OPT:
598 SubmitOPTRecord(ttl, &packet[start + 10], ldata, e_rcode);
599 break;
600 case (int)DnsRtype_DNSKEY:
601 SubmitKeyRecord(&packet[start + 10], ldata);
602 break;
603 case (int)DnsRtype_RRSIG:
604 SubmitRRSIGRecord(&packet[start + 10], ldata);
605 break;
606 case (int)DnsRtype_DS:
607 SubmitDSRecord(&packet[start + 10], ldata);
608 break;
609 case (int)DnsRtype_TLSA:
610 SubmitTLSARecord(&packet[start + 10], ldata);
611 break;
612 default:
613 break;
614 }
615 }
616
617 start += ldata + 10;
618 }
619 }
620
621 return start;
622}

624int DnsStats::SubmitName(uint8_t * packet, uint32_t length, uint32_t start, bool should_tabulate)
625{
626 uint32_t l = 0;
627
628 while (start < length)
629 {
630 l = packet[start];
631
632 if (l == 0)
633 {
634 /* end of parsing*/
635 if (should_tabulate)
636 {
637 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, 0);
638 }
639 start++;
640 break;
641 }
642 else if ((l & 0xC0) == 0xC0)
643 {
644 /* Name compression */
645 if (should_tabulate)
646 {
647 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, 0xC0);
648 }
649
650 if ((start + 2) > length)
651 {
652 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
653 start = length;
654 break;
655 }
656 else
657 {
658 start += 2;
659 break;
660 }
661 }
662 else if (l > 0x3F)
663 {
664 /* found an extension. Don't know how to parse it! */
665 error_flags |= DNS_REGISTRY_ERROR_LABEL(1<<7);
666 if (should_tabulate)
667 {
668 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, l);
669 }
670 start = length;
671 break;
672 }
673 else
674 {
675 /* regular name part. To do: tracking of underscore labels. */
676 if (should_tabulate)
677 {
678 SubmitRegistryNumber(REGISTRY_DNS_LabelType7, 0);
679 }
680 if (start + l + 1 > length)
681 {
682 error_flags |= DNS_REGISTRY_ERROR_FORMAT(1<<8);
683 start = length;
684 break;
685 }
686 else
687 {
688 start += l + 1;
689 }
690 }
691 }
692
693 return start;
694}

696void DnsStats::SubmitOPTRecord(uint32_t flags, uint8_t * content, uint32_t length, uint32_t * e_rcode)
697{
698 uint32_t current_index = 0;
699
700 /* Process the flags and rcodes */
701 if (e_rcode != NULL__null)
702 {
703 *e_rcode = (flags >> 24) & 0xFF;
704 }
705
706 /* Register that EDNS was used */
707 is_using_edns = true;
708
709 /* Register whether the DO bit was set */
710 is_do_flag_set = (flags & (1<<15)) != 0;
711
712 /* Add flags to registration */
713 for (int i = 0; i < 16; i++)
714 {
715 if ((flags & (1 << i)) != 0)
716 {
717 SubmitRegistryNumber(REGISTRY_EDNS_Header_Flags10, 15 - i);
718 }
719 }
720
721 SubmitRegistryNumber(REGISTRY_EDNS_Version_number11, (flags >> 16) & 0xFF);
722
723
724 if (current_index < length) {
725 edns_options = &content[current_index];
726 edns_options_length = length - current_index;
727 }
728 else {
729 edns_options = NULL__null;
730 edns_options_length = 0;
731
732 }
733
734 /* Find the options in the payload */
735 while (current_index + 4 <= length)
736 {
737 uint32_t o_code = (content[current_index] << 8) | content[current_index + 1];
738 uint32_t o_length = (content[current_index+2] << 8) | content[current_index + 3];
739 current_index += 4 + o_length;
740 SubmitRegistryNumber(REGISTRY_EDNS_OPT_CODE8, o_code);
741 }
742}

838int DnsStats::CheckForUnderline(uint8_t * packet, uint32_t length, uint32_t start)
839{
840 uint32_t l = 0;
841 uint32_t name_start = start;
842
843 while (start < length)
844 {
845 l = packet[start];
846
847 if (l == 0)
848 {
849 /* end of parsing*/
850 start++;
851 break;
852 }
853 else if ((l & 0xC0) == 0xC0)
854 {
855 if ((start + 2) > length)
856 {
857 /* error */
858 start = length;
859 break;
860 }
861 else
862 {
863 uint32_t new_start = ((l & 63) << 8) + packet[start + 1];
864
865 if (new_start < name_start)
866 {
867 (void)CheckForUnderline(packet, length, new_start);
868 }
869
870 start += 2;
871 break;
872 }
873 }
874 else if (l > 0x3F)

904 }
905 }

911bool DnsStats::GetTLD(uint8_t * packet, uint32_t length, uint32_t start, uint32_t * offset,
912 uint32_t * previous_offset, int * nb_name_parts)
913{
914 bool ret = true;
915 uint32_t l = 0;
916 uint32_t previous_o = 0;
917 uint32_t previous = 0;
918 uint32_t name_start = start;
919 int nb_parts = 0;
920
921 while (start < length)
922 {
923 l = packet[start];
924
925 if (l == 0)
926 {
927 /* end of parsing*/
928
929 if (previous != 0)
930 {
931 *offset = previous;
932 }
933 else
934 {
935 *offset = start;
936 }
937 break;
938 }
939 else if ((l & 0xC0) == 0xC0)
940 {
941 /* Name compression */
942 if ((start + 2) > length)
943 {
944 ret = false;
945 break;
946 }
947 else
948 {
949 uint32_t new_start = ((l & 63) << 8) + packet[start + 1];
950
951 if (new_start < name_start)
952 {
953 int nb_parts_recursive = 0;
954 uint32_t previous_recursive = 0;
955
956 ret = GetTLD(packet, length, new_start, offset, &previous_recursive, &nb_parts_recursive);
957 if (previous_recursive != 0) {
958 previous_o = previous_recursive;
959 }
960 else {
961 previous_o = previous;
962 }
963 nb_parts += nb_parts_recursive;
964 }
965 else
966 {
967 ret = false;
968 }
969 break;
970 }
971 }
972 else if (l > 0x3F)
973 {
974 /* Unexpected name part */
975 ret = false;
976 break;
977 }
978 else
979 {
980 if (start + l + 1 > length)
981 {
982 /* malformed name part */
983 ret = false;
984 break;
985 }
986 else
987 {
988 nb_parts++;
989 previous_o = previous;
990 previous = start;
991 start += l + 1;
992 }
993 }
994 }
995
996 if (nb_name_parts != NULL__null) {
997 *nb_name_parts = nb_parts;
998 }
999
1000 if (previous_offset != NULL__null) {
1001 *previous_offset = previous_o;
1002 }
1003
1004 return ret;
1005}

1007int DnsStats::GetDnsName(uint8_t * packet, uint32_t length, uint32_t start,
1008 uint8_t * name, size_t name_max, size_t * name_length)
1009{
1010 uint32_t l = 0;
1011 uint32_t name_start = start;
1012 uint32_t start_next = 0;
1013 size_t name_index = 0;
1014
1015 while (start <length && name_index < name_max) {
1016 l = packet[start];
1017
1018 if (l == 0)
1019 {
1020 /* end of parsing*/
1021 start++;
1022
1023 if (start_next == 0) {
1024 start_next = start;
1025 }
1026 break;
1027 }
1028 else if ((l & 0xC0) == 0xC0)
1029 {
1030 if ((start + 2) > length)
1031 {
1032 /* error */
1033 start_next = length;
1034 break;
1035 }
1036 else
1037 {
1038 uint32_t new_start = ((l & 63) << 8) + packet[start + 1];
1039
1040 if (new_start < name_start)
1041 {
1042 if (start_next == 0) {
1043 start_next = start + 2;
1044 }
1045 start = new_start;
1046 } else {
1047 /* Basic restriction to avoid name decoding loops */
1048 name_index = 0;
1049 start_next = length;
1050 break;
1051 }
1052 }
1053 }
1054 else if (l > 0x3F)
1055 {
1056 /* found an extension. Don't know how to parse it! */
1057 name_index = 0;
1058 start_next = length;
1059 break;
1060 }
1061 else
1062 {
1063 /* add a label to the name. */
1064 if (start + l + 1 > length ||
1065 name_index + l + 2 > name_max)
1066 {
1067 /* format error */
1068 name_index = 0;
1069 start_next = length;
1070 break;
1071 }
1072 else
1073 {
1074 uint32_t flags;
1075 if (name_index > 0 && name_index+1 < name_max) {
1076 name[name_index++] = '.';
1077 }
1078
1079 name_index += NormalizeNamePart(l, &packet[start + 1], name + name_index, name_max - name_index, &flags);
1080
1081 start += l + 1;
1082 }
1083 }
1084 }
1085
1086 name[name_index] = 0;
1087
1088 *name_length = name_index;
1089
1090 return start_next;
1091
1092}

1094int DnsStats::CompareDnsName(uint8_t * packet, uint32_t length, uint32_t start1, uint32_t start2)
1095{
1096 bool ret = false;
1097
1098 while (start1 < length && start2 < length) {
1099 if (start1 == start2) {
1100 ret = true;
1101 break;
1102 }
1103 else if ((packet[start1] & 0xC0) == 0xC0)
1104 {
1105 start1 = ((packet[start1] & 63) << 8) + packet[start1 + 1];
1106 }
1107 else if ((packet[start2] & 0xC0) == 0xC0)
1108 {
1109 start2 = ((packet[start2] & 63) << 8) + packet[start2 + 1];
1110 }
1111 else if (packet[start1] > 0x3f || packet[start2] > 0x3f)
1112 {
1113 break;
1114 }
1115 else if (packet[start1] != packet[start2])
1116 {
1117 break;
1118 }
1119 else
1120 {
1121 uint32_t l = packet[start1];
1122 start1++;
1123 start2++;
1124
1125 if (l == 0)
1126 {
1127 ret = true;
1128 break;
1129 }
1130 else if (start1 + l > length || start2 + l >= length)
1131 {
1132 break;
1133 }
1134 else
1135 {
1136 bool cmp = true;
1137 for (uint32_t i = 0; cmp && i < l; i++) {
1138 uint8_t c1 = packet[start1 + i];
1139 uint8_t c2 = packet[start2 + i];
1140
1141 cmp = (c1 == c2 || (c1 >= 'a' && c1 <= 'z' && (c1 + 'A' - 'a') == c2) || (c1 >= 'A' && c1 <= 'Z' && (c1 + 'a' - 'A') == c2));
1142 }
1143
1144 if (!cmp) {
1145 break;
1146 }
1147 start1 += l;
1148 start2 += l;
1149 }
1150 }
1151 }
1152
1153 return ret;

1230size_t DnsStats::NormalizeNamePart(uint32_t length, uint8_t * value,
1231 uint8_t * normalized, size_t normalized_max, uint32_t * flags)
1232{
1233 bool has_letter = false;
1234 bool has_number = false;
1235 bool has_special = false;
1236 bool has_dash = false;
1237 bool has_non_ascii = false;
1238 size_t normal_length = 0;
1239
1240 for (uint32_t i = 0; i < length && normal_length + 1 < normalized_max; i++)
1241 {
1242 uint8_t c = value[i];
1243 bool need_escape = false;
1244
1245 if (c >= 'A' && c <= 'Z')
1246 {
1247 has_letter = true;
1248 }
1249 else if (c >= 'a' && c <= 'z')
1250 {
1251 has_letter = true;
1252 }
1253 else if (c >= '0' && c <= '9')
1254 {
1255 has_number = true;
1256 }
1257 else if (c == '-' || c == '_')
1258 {
1259 has_dash = true;
1260 }
1261 else if (c == 127 || c < ' ')
1262 {
1263 need_escape = true;
1264 has_special = true;
1265 }
1266 else if (c > 127) {
1267 need_escape = true;
1268 has_non_ascii = true;
1269 }
1270 else if (c == ' ')
1271 {
1272 need_escape = (i == 0 || i == (length - 1));
1273 has_special = true;
1274 }
1275 else
1276 {
1277 has_special = true;
1278 need_escape = (c == '.');
1279 }
1280 if (need_escape) {
1281 if (normal_length + 5 < normalized_max) {
1282 int dec[3];
1283 dec[0] = c / 100;
1284 dec[1] = (c % 100) / 10;
1285 dec[2] = c % 10;
1286
1287 normalized[normal_length++] = '\\';
1288 for (int x = 0; x < 3; x++) {
1289 normalized[normal_length++] = '0' + dec[x];
1290 }
1291 }
1292 else {
1293 normalized[normal_length++] = '!';
1294 }
1295 }
1296 else {
1297 normalized[normal_length++] = c;
1298 }
1299 }
1300 normalized[normal_length] = 0;
1301
1302 if (flags != NULL__null)
1303 {
1304 *flags = 0;
1305
1306 if (has_non_ascii)
1307 {
1308 *flags += 1;
1309 }
1310 else if (has_special)
1311 {
1312 *flags += 2;
1313 }
1314
1315 if (has_letter)
1316 {
1317 *flags += 64;
1318 }
1319 if (has_number)
1320 {
1321 *flags += 128;
1322 }
1323 if (has_dash)
1324 {
1325 *flags += 256;
1326 }
1327 }
1328
1329 return(normal_length);
1330}
